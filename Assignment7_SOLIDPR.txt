public class LedgerEntry {}



public class Ledger {

    private Set<LedgerEntry> entries;



    public void addEntry(final LedgerEntry entry) {

        if (entry != null) {

            if (entries == null) {

                entries = new HashSet<>();

            }



            entries.add(entry);

        }

    }



    public void getEntries() {

        return entries == null ? null

                               : new HashSet<>(entries);

    }

}
